Table contest {
  id integer [primary key]
  categoryID integer  
}

Table users {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp  
}

Table photo {
  id integer [primary key]
  title varchar
  body text [note: 'Content of the post']
  user_id integer
  status varchar
  created_at timestamp
}

Table ParticipatingPhotos {
  id integer [primary key]  
  photoId list
  likes integer
  photoLimit integer
  categoryId integer
}

Table UserParticipation {
  id integer [primary key]
  contestID integer
  userId integer
  status string
}

Table JuryLikes {  
  userId integer
  participatingPhotoId integer
}

Table Category {
  id integer [primary key]
  participatingPhotoId integer
}

Table usercontests {  
  
  userId integer
  contestId integer
}
